#include<stdio.h>#include<string.h>int validate_credentials(char* username, char* password) {    if (strcmp(username, password) == 0) {        return 1; // 验证通过      }    else {        return 0; // 验证失败      }}int main() {    char username[20];    char password[20];    int i;    int max_retries = 3;    int retries = 0;    while (retries < max_retries) {        printf("请输入用户名：");        scanf("%s", username);        printf("请输入密码：");        scanf("%s", password);        if (validate_credentials(username, password)) {            process_data();            break;        }        else {            printf("用户名或密码错误，请重新输入。\n");            retries++;        }    }    if (retries == max_retries) {        printf("程序退出。\n");    }    else {        printf("按R键重新执行程序，按Q键退出程序。\n");    }    return 0;}
